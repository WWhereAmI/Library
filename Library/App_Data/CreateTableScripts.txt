CREATE TABLE "BookAudit" (
	"Id"	INTEGER,
	"BookId"	INTEGER NOT NULL UNIQUE,
	"ClientId"	INTEGER NOT NULL,
	PRIMARY KEY("Id" AUTOINCREMENT)
)

CREATE TABLE "Client" (
	"Id"	INTEGER,
	"Name"	TEXT,
	"Family"	TEXT,
	PRIMARY KEY("Id" AUTOINCREMENT)
)


CREATE TABLE "Book" (
	"Id"	INTEGER,
	"Title"	TEXT NOT NULL,
	"Description"	TEXT,
	"Author"	TEXT NOT NULL,
	"Status"	INTEGER NOT NULL DEFAULT 1,
	PRIMARY KEY("Id" AUTOINCREMENT)
)
